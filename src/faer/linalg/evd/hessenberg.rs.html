7/// hessenberg factorization tuning parameters
8#[derive(Copy, Clone, Debug)]
9pub struct HessenbergParams {
10	/// threshold at which parallelism should be disabled
11	pub par_threshold: usize,
12	/// threshold at which parallelism should be disabled
13	pub blocking_threshold: usize,
14
15	#[doc(hidden)]
16	pub non_exhaustive: NonExhaustive,
17}

19impl<T: ComplexField> Auto<T> for HessenbergParams {
20	fn auto() -> Self {
21		Self {
22			par_threshold: 192 * 256,
23			blocking_threshold: 256 * 256,
24			non_exhaustive: NonExhaustive(()),
25		}
26	}
27}

29/// computes the size and alignment of the workspace required to compute a matrix's hessenberg
30/// decomposition
31pub fn hessenberg_in_place_scratch<T: ComplexField>(dim: usize, blocksize: usize, par: Par, params: Spec<HessenbergParams, T>) -> StackReq {
32	let params = params.config;
33	let _ = par;
34	let n = dim;
35	if n * n < params.blocking_threshold {
36		StackReq::any_of(&[StackReq::all_of(&[
37			temp_mat_scratch::<T>(n, 1).array(3),
38			temp_mat_scratch::<T>(n, par.degree()),
39		])])
40	} else {
41		StackReq::all_of(&[
42			temp_mat_scratch::<T>(n, blocksize),
43			temp_mat_scratch::<T>(blocksize, 1),
44			StackReq::any_of(&[
45				StackReq::all_of(&[temp_mat_scratch::<T>(n, 1), temp_mat_scratch::<T>(n, par.degree())]),
46				temp_mat_scratch::<T>(n, blocksize),
47			]),
48		])
49	}
50}

52#[math]
53fn hessenberg_fused_op_simd<T: ComplexField>(
54	A: MatMut<'_, T, usize, usize, ContiguousFwd>,
55
56	l_out: RowMut<'_, T, usize>,
57	r_out: ColMut<'_, T, usize, ContiguousFwd>,
58	l_in: RowRef<'_, T, usize, ContiguousFwd>,
59	r_in: ColRef<'_, T, usize>,
60
61	l0: ColRef<'_, T, usize, ContiguousFwd>,
62	l1: ColRef<'_, T, usize, ContiguousFwd>,
63	r0: RowRef<'_, T, usize>,
64	r1: RowRef<'_, T, usize>,
65	align: usize,
66) {
67	struct Impl<'a, 'M, 'N, T: ComplexField> {
68		A: MatMut<'a, T, Dim<'M>, Dim<'N>, ContiguousFwd>,
69
70		l_out: RowMut<'a, T, Dim<'N>>,
71		r_out: ColMut<'a, T, Dim<'M>, ContiguousFwd>,
72		l_in: RowRef<'a, T, Dim<'M>, ContiguousFwd>,
73		r_in: ColRef<'a, T, Dim<'N>>,
74
75		l0: ColRef<'a, T, Dim<'M>, ContiguousFwd>,
76		l1: ColRef<'a, T, Dim<'M>, ContiguousFwd>,
77		r0: RowRef<'a, T, Dim<'N>>,
78		r1: RowRef<'a, T, Dim<'N>>,
79		align: usize,
80	}
81
82	impl<'a, 'M, 'N, T: ComplexField> pulp::WithSimd for Impl<'a, 'M, 'N, T> {
83		type Output = ();
84
85		#[inline(always)]
86		fn with_simd<S: pulp::Simd>(self, simd: S) -> Self::Output {
87			let Self {
88				mut A,
89				mut l_out,
90				mut r_out,
91				l_in,
92				r_in,
93				l0,
94				l1,
95				r0,
96				r1,
97				align,
98			} = self;
99
100			let (m, n) = A.shape();
101
102			let simd = SimdCtx::<T, S>::new_align(T::simd_ctx(simd), m, align);
103
104			{
105				let (head, body, tail) = simd.indices();
106				if let Some(i) = head {
107					simd.write(r_out.rb_mut(), i, simd.zero());
108				}
109				for i in body {
110					simd.write(r_out.rb_mut(), i, simd.zero());
111				}
112				if let Some(i) = tail {
113					simd.write(r_out.rb_mut(), i, simd.zero());
114				}
115			}
116
117			let (head, body4, body1, tail) = simd.batch_indices::<4>();
118
119			let l_in = l_in.transpose();
120
121			for j in n.indices() {
122				let mut A = A.rb_mut().col_mut(j);
123				let r_in = simd.splat(r_in.at(j));
124				let r0 = simd.splat(&(-r0[j]));
125				let r1 = simd.splat(&(-r1[j]));
126
127				let mut acc0 = simd.zero();
128				let mut acc1 = simd.zero();
129				let mut acc2 = simd.zero();
130				let mut acc3 = simd.zero();
131
132				if let Some(i0) = head {
133					let mut a0 = simd.read(A.rb(), i0);
134					a0 = simd.mul_add(simd.read(l0, i0), r0, a0);
135					a0 = simd.conj_mul_add(r1, simd.read(l1, i0), a0);
136					simd.write(A.rb_mut(), i0, a0);
137					acc0 = simd.conj_mul_add(simd.read(l_in, i0), a0, acc0);
138					let tmp = simd.read(r_out.rb(), i0);
139					simd.write(r_out.rb_mut(), i0, simd.mul_add(a0, r_in, tmp));
140				}
141				for [i0, i1, i2, i3] in body4.clone() {
142					{
143						let mut a0 = simd.read(A.rb(), i0);
144						a0 = simd.mul_add(simd.read(l0, i0), r0, a0);
145						a0 = simd.conj_mul_add(r1, simd.read(l1, i0), a0);
146						simd.write(A.rb_mut(), i0, a0);
147						acc0 = simd.conj_mul_add(simd.read(l_in, i0), a0, acc0);
148						let tmp = simd.read(r_out.rb(), i0);
149						simd.write(r_out.rb_mut(), i0, simd.mul_add(a0, r_in, tmp));
150					}
151					{
152						let mut a1 = simd.read(A.rb(), i1);
153						a1 = simd.mul_add(simd.read(l0, i1), r0, a1);
154						a1 = simd.conj_mul_add(r1, simd.read(l1, i1), a1);
155						simd.write(A.rb_mut(), i1, a1);
156						acc1 = simd.conj_mul_add(simd.read(l_in, i1), a1, acc1);
157						let tmp = simd.read(r_out.rb(), i1);
158						simd.write(r_out.rb_mut(), i1, simd.mul_add(a1, r_in, tmp));
159					}
160					{
161						let mut a2 = simd.read(A.rb(), i2);
162						a2 = simd.mul_add(simd.read(l0, i2), r0, a2);
163						a2 = simd.conj_mul_add(r1, simd.read(l1, i2), a2);
164						simd.write(A.rb_mut(), i2, a2);
165						acc2 = simd.conj_mul_add(simd.read(l_in, i2), a2, acc2);
166						let tmp = simd.read(r_out.rb(), i2);
167						simd.write(r_out.rb_mut(), i2, simd.mul_add(a2, r_in, tmp));
168					}
169					{
170						let mut a3 = simd.read(A.rb(), i3);
171						a3 = simd.mul_add(simd.read(l0, i3), r0, a3);
172						a3 = simd.conj_mul_add(r1, simd.read(l1, i3), a3);
173						simd.write(A.rb_mut(), i3, a3);
174						acc3 = simd.conj_mul_add(simd.read(l_in, i3), a3, acc3);
175						let tmp = simd.read(r_out.rb(), i3);
176						simd.write(r_out.rb_mut(), i3, simd.mul_add(a3, r_in, tmp));
177					}
178				}
179				for i0 in body1.clone() {
180					let mut a0 = simd.read(A.rb(), i0);
181					a0 = simd.mul_add(simd.read(l0, i0), r0, a0);
182					a0 = simd.conj_mul_add(r1, simd.read(l1, i0), a0);
183					simd.write(A.rb_mut(), i0, a0);
184					acc0 = simd.conj_mul_add(simd.read(l_in, i0), a0, acc0);
185					let tmp = simd.read(r_out.rb(), i0);
186					simd.write(r_out.rb_mut(), i0, simd.mul_add(a0, r_in, tmp));
187				}
188				if let Some(i0) = tail {
189					let mut a0 = simd.read(A.rb(), i0);
190					a0 = simd.mul_add(simd.read(l0, i0), r0, a0);
191					a0 = simd.conj_mul_add(r1, simd.read(l1, i0), a0);
192					simd.write(A.rb_mut(), i0, a0);
193					acc0 = simd.conj_mul_add(simd.read(l_in, i0), a0, acc0);
194					let tmp = simd.read(r_out.rb(), i0);
195					simd.write(r_out.rb_mut(), i0, simd.mul_add(a0, r_in, tmp));
196				}
197
198				acc0 = simd.add(acc0, acc1);
199				acc2 = simd.add(acc2, acc3);
200				acc0 = simd.add(acc0, acc2);
201
202				let l_out = l_out.rb_mut().at_mut(j);
203				*l_out = simd.reduce_sum(acc0);
204			}
205		}
206	}
207
208	with_dim!(M, A.nrows());
209	with_dim!(N, A.ncols());
210
211	dispatch!(
212		Impl {
213			A: A.as_shape_mut(M, N),
214			l_out: l_out.as_col_shape_mut(N),
215			r_out: r_out.as_row_shape_mut(M),
216			l_in: l_in.as_col_shape(M),
217			r_in: r_in.as_row_shape(N),
218			l0: l0.as_row_shape(M),
219			l1: l1.as_row_shape(M),
220			r0: r0.as_col_shape(N),
221			r1: r1.as_col_shape(N),
222			align,
223		},
224		Impl,
225		T
226	)
227}

229#[math]
230fn hessenberg_fused_op_fallback<T: ComplexField>(
231	A: MatMut<'_, T>,
232
233	l_out: RowMut<'_, T>,
234	r_out: ColMut<'_, T>,
235	l_in: RowRef<'_, T>,
236	r_in: ColRef<'_, T>,
237
238	l0: ColRef<'_, T>,
239	l1: ColRef<'_, T>,
240	r0: RowRef<'_, T>,
241	r1: RowRef<'_, T>,
242) {
243	let mut A = A;
244
245	matmul(A.rb_mut(), Accum::Add, l0.as_mat(), r0.as_mat(), -one::<T>(), Par::Seq);
246	matmul(A.rb_mut(), Accum::Add, l1.as_mat(), r1.as_mat().conjugate(), -one::<T>(), Par::Seq);
247
248	matmul(r_out.as_mat_mut(), Accum::Replace, A.rb(), r_in.as_mat(), one(), Par::Seq);
249	matmul(l_out.as_mat_mut(), Accum::Replace, l_in.as_mat().conjugate(), A.rb(), one(), Par::Seq);
250}

252fn hessenberg_fused_op<T: ComplexField>(
253	A: MatMut<'_, T>,
254
255	l_out: RowMut<'_, T>,
256	r_out: ColMut<'_, T>,
257	l_in: RowRef<'_, T>,
258	r_in: ColRef<'_, T>,
259
260	l0: ColRef<'_, T>,
261	l1: ColRef<'_, T>,
262	r0: RowRef<'_, T>,
263	r1: RowRef<'_, T>,
264	align: usize,
265) {
266	let mut A = A;
267	let mut r_out = r_out;
268
269	if try_const! { T::SIMD_CAPABILITIES.is_simd() } {
270		if let (Some(A), Some(r_out), Some(l_in), Some(l0), Some(l1)) = (
271			A.rb_mut().try_as_col_major_mut(),
272			r_out.rb_mut().try_as_col_major_mut(),
273			l_in.try_as_row_major(),
274			l0.try_as_col_major(),
275			l1.try_as_col_major(),
276		) {
277			hessenberg_fused_op_simd(A, l_out, r_out, l_in, r_in, l0, l1, r0, r1, align);
278		} else {
279			hessenberg_fused_op_fallback(A, l_out, r_out, l_in, r_in, l0, l1, r0, r1);
280		}
281	} else {
282		hessenberg_fused_op_fallback(A, l_out, r_out, l_in, r_in, l0, l1, r0, r1);
283	}
284}

286#[math]
287fn hessenberg_rearranged_unblocked<T: ComplexField>(A: MatMut<'_, T>, H: MatMut<'_, T>, par: Par, stack: &mut MemStack, params: HessenbergParams) {
288	assert!(all(A.nrows() == A.ncols(), H.ncols() == A.ncols().saturating_sub(1)));
289
290	let n = A.nrows();
291	let b = H.nrows();
292
293	if n == 0 {
294		return;
295	}
296
297	let mut A = A;
298	let mut H = H;
299	let mut par = par;
300
301	{
302		let mut H = H.rb_mut().row_mut(0);
303		let (mut y, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
304		let (mut z, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
305		let (mut v, stack) = unsafe { temp_mat_uninit(n, 1, stack) };
306		let (mut w, _) = unsafe { temp_mat_uninit(n, par.degree(), stack) };
307
308		let mut y = y.as_mat_mut().col_mut(0).transpose_mut();
309		let mut z = z.as_mat_mut().col_mut(0);
310		let mut v = v.as_mat_mut().col_mut(0).transpose_mut();
311		let mut w = w.as_mat_mut();
312
313		for k in 0..n {
314			let (_, A01, A10, A11) = A.rb_mut().split_at_mut(k, k);
315
316			let (_, mut A02) = A01.split_first_col_mut().unwrap();
317			let (_, A20) = A10.split_first_row_mut().unwrap();
318			let (mut A11, A12, A21, mut A22) = A11.split_at_mut(1, 1);
319
320			let mut A12 = A12.row_mut(0);
321			let mut A21 = A21.col_mut(0);
322
323			let A11 = &mut A11[(0, 0)];
324
325			let (y1, mut y2) = y.rb_mut().split_at_col_mut(k).1.split_at_col_mut(1);
326			let y1 = copy(y1[0]);
327
328			let (z1, mut z2) = z.rb_mut().split_at_row_mut(k).1.split_at_row_mut(1);
329			let z1 = copy(z1[0]);
330
331			let (_, mut v2) = v.rb_mut().split_at_col_mut(k).1.split_at_col_mut(1);
332			let (mut w0, w12) = w.rb_mut().split_at_row_mut(k);
333			let (_, mut w2) = w12.split_at_row_mut(1);
334
335			if k > 0 {
336				let p = k - 1;
337				let u2 = A20.rb().col(p);
338
339				*A11 = *A11 - y1 - z1;
340				z!(&mut A12, &y2, u2.rb().transpose()).for_each(|uz!(a, y, u)| *a = *a - *y - z1 * conj(*u));
341				z!(&mut A21, &u2, &z2).for_each(|uz!(a, u, z)| *a = *a - *u * y1 - *z);
342			}
343
344			{
345				let n = n - k - 1;
346				if n * n < params.par_threshold {
347					par = Par::Seq;
348				}
349			}
350
351			if k + 1 == n {
352				break;
353			}
354
355			let beta;
356			let tau_inv;
357			{
358				let (mut A11, mut A21) = A21.rb_mut().split_at_row_mut(1);
359				let A11 = &mut A11[0];
360
361				let (tau, _) = householder::make_householder_in_place(A11, A21.rb_mut());
362				tau_inv = recip(real(tau));
363				beta = copy(*A11);
364				*A11 = one();
365
366				H[k] = tau;
367			}
368
369			let x2 = A21.rb();
370
371			if k > 0 {
372				let p = k - 1;
373				let u2 = A20.rb().col(p);
374				hessenberg_fused_op(
375					A22.rb_mut(),
376					v2.rb_mut(),
377					w2.rb_mut().col_mut(0),
378					x2.transpose(),
379					x2,
380					u2,
381					z2.rb(),
382					y2.rb(),
383					u2.transpose(),
384					simd_align(k + 1),
385				);
386				y2.copy_from(v2.rb());
387				z2.copy_from(w2.rb().col(0));
388			} else {
389				matmul(z2.rb_mut().as_mat_mut(), Accum::Replace, A22.rb(), x2.as_mat(), one(), par);
390				matmul(y2.rb_mut().as_mat_mut(), Accum::Replace, x2.adjoint().as_mat(), A22.rb(), one(), par);
391			}
392
393			let u2 = x2;
394
395			let b = mul_real(
396				mul_pow2(dot::inner_prod(u2.rb().transpose(), Conj::Yes, z2.rb(), Conj::No), from_f64(0.5)),
397				tau_inv,
398			);
399			z!(&mut y2, u2.transpose()).for_each(|uz!(y, u)| *y = mul_real(*y - b * conj(*u), tau_inv));
400			z!(&mut z2, u2).for_each(|uz!(z, u)| *z = mul_real(*z - b * *u, tau_inv));
401
402			let dot = mul_real(dot::inner_prod(A12.rb(), Conj::No, u2.rb(), Conj::No), tau_inv);
403			z!(&mut A12, u2.transpose()).for_each(|uz!(a, u)| *a = *a - dot * conj(u));
404
405			matmul(w0.rb_mut().col_mut(0).as_mat_mut(), Accum::Replace, A02.rb(), u2.as_mat(), one(), par);
406			matmul(
407				A02.rb_mut(),
408				Accum::Add,
409				w0.rb().col(0).as_mat(),
410				u2.adjoint().as_mat(),
411				-from_real::<T>(&tau_inv),
412				par,
413			);
414
415			A21[0] = beta;
416		}
417	}
418
419	if n > 0 {
420		let n = n - 1;
421		let A = A.rb().submatrix(1, 0, n, n);
422		let mut H = H.rb_mut().subcols_mut(0, n);
423
424		let mut j = 0;
425		while j < n {
426			let b = Ord::min(b, n - j);
427
428			let mut H = H.rb_mut().submatrix_mut(0, j, b, b);
429
430			for k in 0..b {
431				H[(k, k)] = copy(H[(0, k)]);
432			}
433
434			householder::upgrade_householder_factor(H.rb_mut(), A.submatrix(j, j, n - j, b), b, 1, par);
435			j += b;
436		}
437	}
438}

440#[math]
441fn hessenberg_gqvdg_unblocked<T: ComplexField>(
442	A: MatMut<'_, T>,
443	Z: MatMut<'_, T>,
444	H: MatMut<'_, T>,
445	beta: ColMut<'_, T>,
446	par: Par,
447	stack: &mut MemStack,
448	params: HessenbergParams,
449) {
450	let n = A.nrows();
451	let b = H.nrows();
452	let mut A = A;
453	let mut H = H;
454	let mut Z = Z;
455	_ = params;
456
457	let (mut x, _) = unsafe { temp_mat_uninit(n, 1, stack) };
458	let mut x = x.as_mat_mut().col_mut(0);
459	let mut beta = beta;
460
461	for k in 0..b {
462		let mut x0 = x.rb_mut().subrows_mut(0, k);
463		let (T00, T01, _, T11) = H.rb_mut().split_at_mut(k, k);
464		let (mut T01, _) = T01.split_first_col_mut().unwrap();
465		let (mut T11, _, _, _) = T11.split_at_mut(1, 1);
466
467		let T11 = &mut T11[(0, 0)];
468
469		let (U0, A12) = A.rb_mut().split_at_col_mut(k);
470		let (mut A1, A2) = A12.split_first_col_mut().unwrap();
471
472		let (Z0, Z12) = Z.rb_mut().split_at_col_mut(k);
473		let (mut Z1, _) = Z12.split_first_col_mut().unwrap();
474
475		let U0 = U0.rb();
476		let Z0 = Z0.rb();
477		let T00 = T00.rb();
478
479		let (U00, U10) = U0.split_at_row(k);
480		let (U10, U20) = U10.split_first_row().unwrap();
481
482		x0.copy_from(U10.adjoint());
483		triangular_solve::solve_upper_triangular_in_place(T00, x0.rb_mut().as_mat_mut(), par);
484		matmul::matmul(A1.rb_mut().as_mat_mut(), Accum::Add, Z0, x0.rb().as_mat(), -one::<T>(), par);
485
486		let (mut A01, A11) = A1.rb_mut().split_at_row_mut(k);
487		let (mut A11, mut A21) = A11.split_at_row_mut(1);
488		let A11 = &mut A11[0];
489
490		{
491			matmul::triangular::matmul(
492				x0.rb_mut().as_mat_mut(),
493				BlockStructure::Rectangular,
494				Accum::Replace,
495				U00.adjoint(),
496				BlockStructure::StrictTriangularUpper,
497				A01.rb().as_mat(),
498				BlockStructure::Rectangular,
499				one(),
500				par,
501			);
502			z!(x0.rb_mut(), U10.transpose()).for_each(|uz!(x, u)| *x = *x + *A11 * conj(*u));
503			matmul::matmul(x0.rb_mut().as_mat_mut(), Accum::Add, U20.adjoint(), A21.rb().as_mat(), one(), par);
504		}
505		{
506			triangular_solve::solve_lower_triangular_in_place(T00.adjoint(), x0.rb_mut().as_mat_mut(), par);
507		}
508		{
509			matmul::triangular::matmul(
510				A01.rb_mut().as_mat_mut(),
511				BlockStructure::Rectangular,
512				Accum::Add,
513				U00,
514				BlockStructure::StrictTriangularLower,
515				x0.rb().as_mat(),
516				BlockStructure::Rectangular,
517				-one::<T>(),
518				par,
519			);
520			*A11 = *A11 - dot::inner_prod(U10, Conj::No, x0.rb(), Conj::No);
521			matmul::matmul(A21.rb_mut().as_mat_mut(), Accum::Add, U20, x0.rb().as_mat(), -one::<T>(), par);
522		}
523
524		if k + 1 < n {
525			let (mut A11, mut A21) = A21.rb_mut().split_at_row_mut(1);
526			let A11 = &mut A11[0];
527
528			let (tau, _) = householder::make_householder_in_place(A11, A21.rb_mut());
529
530			beta[k] = copy(A11);
531			*A11 = one();
532			*T11 = tau;
533		} else {
534			*T11 = infinity();
535		}
536
537		matmul::matmul(Z1.rb_mut().as_mat_mut(), Accum::Replace, A2.rb(), A21.rb().as_mat(), one(), par);
538
539		matmul::matmul(T01.rb_mut().as_mat_mut(), Accum::Replace, U20.adjoint(), A21.rb().as_mat(), one(), par);
540	}
541}

543/// computes a matrix $A$'s hessenberg decomposition such that $A = Q H Q^H$
544///
545/// $H$ is a hessenberg matrix stored in the upper triangular half of $A$ (plus the subdiagonal)
546///
547/// $Q$ is a sequence of householder reflections stored in the unit lower triangular half of $A$
548/// (excluding the diagonal), with the householder coefficients being stored in `householder`
549#[track_caller]
550pub fn hessenberg_in_place<T: ComplexField>(
551	A: MatMut<'_, T>,
552	householder: MatMut<'_, T>,
553	par: Par,
554	stack: &mut MemStack,
555	params: Spec<HessenbergParams, T>,
556) {
557	let params = params.config;
558	assert!(all(A.nrows() == A.ncols(), householder.ncols() == A.ncols().saturating_sub(1)));
559
560	let n = A.nrows().unbound();
561
562	if n * n < params.blocking_threshold {
563		hessenberg_rearranged_unblocked(A, householder, par, stack, params);
564	} else {
565		hessenberg_gqvdg_blocked(A, householder, par, stack, params);
566	}
567}
568
569#[math]
570fn hessenberg_gqvdg_blocked<T: ComplexField>(A: MatMut<'_, T>, H: MatMut<'_, T>, par: Par, stack: &mut MemStack, params: HessenbergParams) {
571	let n = A.nrows();
572	let b = H.nrows();
573	let mut A = A;
574	let mut H = H;
575	let (mut Z, stack) = unsafe { temp_mat_uninit(n, b, stack) };
576	let mut Z = Z.as_mat_mut();
577
578	let mut j = 0;
579	while j < n {
580		let bs = Ord::min(b, n - j);
581		let bs_u = Ord::min(bs, n - j - 1);
582
583		let (mut beta, stack) = unsafe { temp_mat_uninit(bs, 1, stack) };
584		let mut beta = beta.as_mat_mut().col_mut(0);
585
586		{
587			let mut T11 = H.rb_mut().submatrix_mut(0, j, bs_u, bs_u);
588			{
589				let A11 = A.rb_mut().submatrix_mut(j, j, n - j, n - j);
590				let Z1 = Z.rb_mut().submatrix_mut(j, 0, n - j, bs);
591
592				hessenberg_gqvdg_unblocked(A11, Z1, T11.rb_mut(), beta.rb_mut(), par, stack, params);
593			}
594
595			let (mut X, _) = unsafe { temp_mat_uninit(n, bs_u, stack) };
596			let mut X = X.as_mat_mut();
597
598			let (mut X0, X12) = X.rb_mut().split_at_row_mut(j);
599			let (_, mut X2) = X12.split_at_row_mut(bs_u);
600
601			let (_, Z12) = Z.rb_mut().subcols_mut(0, bs_u).split_at_row_mut(j);
602			let (mut Z1, mut Z2) = Z12.split_at_row_mut(bs_u);
603
604			let (_, A01, _, A11) = A.rb_mut().split_at_mut(j, j);
605			let (mut A01, mut A02) = A01.split_at_col_mut(bs_u);
606			let (A11, mut A12, A21, mut A22) = A11.split_at_mut(bs_u, bs_u);
607
608			let U1 = A11.rb();
609			let U2 = A21.rb();
610
611			let T1 = T11.rb();
612
613			matmul::triangular::matmul(
614				X0.rb_mut(),
615				BlockStructure::Rectangular,
616				Accum::Replace,
617				A01.rb(),
618				BlockStructure::Rectangular,
619				U1,
620				BlockStructure::StrictTriangularLower,
621				one(),
622				par,
623			);
624			matmul::matmul(X0.rb_mut(), Accum::Add, A02.rb(), U2, one(), par);
625
626			triangular_solve::solve_lower_triangular_in_place(T1.transpose(), X0.rb_mut().transpose_mut(), par);
627
628			matmul::triangular::matmul(
629				A01.rb_mut(),
630				BlockStructure::Rectangular,
631				Accum::Add,
632				X0.rb(),
633				BlockStructure::Rectangular,
634				U1.adjoint(),
635				BlockStructure::StrictTriangularUpper,
636				-one::<T>(),
637				par,
638			);
639			matmul::matmul(A02.rb_mut(), Accum::Add, X0.rb(), U2.adjoint(), -one::<T>(), par);
640
641			triangular_solve::solve_lower_triangular_in_place(T1.transpose(), Z1.rb_mut().transpose_mut(), par);
642			triangular_solve::solve_lower_triangular_in_place(T1.transpose(), Z2.rb_mut().transpose_mut(), par);
643
644			matmul::matmul(A12.rb_mut(), Accum::Add, Z1.rb(), U2.adjoint(), -one::<T>(), par);
645			matmul::matmul(A22.rb_mut(), Accum::Add, Z2.rb(), U2.adjoint(), -one::<T>(), par);
646
647			let mut X = X2.rb_mut().transpose_mut();
648
649			matmul::triangular::matmul(
650				X.rb_mut(),
651				BlockStructure::Rectangular,
652				Accum::Replace,
653				U1.adjoint(),
654				BlockStructure::StrictTriangularUpper,
655				A12.rb(),
656				BlockStructure::Rectangular,
657				one(),
658				par,
659			);
660			matmul::matmul(X.rb_mut(), Accum::Add, U2.adjoint(), A22.rb(), one(), par);
661
662			triangular_solve::solve_lower_triangular_in_place(T1.adjoint(), X.rb_mut(), par);
663
664			matmul::triangular::matmul(
665				A12.rb_mut(),
666				BlockStructure::Rectangular,
667				Accum::Add,
668				U1,
669				BlockStructure::StrictTriangularLower,
670				X.rb(),
671				BlockStructure::Rectangular,
672				-one::<T>(),
673				par,
674			);
675			matmul::matmul(A22.rb_mut(), Accum::Add, U2, X.rb(), -one::<T>(), par);
676		}
677
678		let n = n - j;
679		let mut A = A.rb_mut().submatrix_mut(j, j, n, bs);
680		for k in 0..bs {
681			if k + 1 < n {
682				A[(k + 1, k)] = copy(beta[k]);
683			}
684		}
685
686		j += bs;
687	}
688}
