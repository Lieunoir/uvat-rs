8/// immutable view over a row vector, similar to an immutable reference to a strided
9/// [prim@slice]
10///
11/// # note
12///
13/// unlike a slice, the data pointed to by `RowRef<'_, T>` is allowed to be partially or fully
14/// uninitialized under certain conditions. in this case, care must be taken to not perform any
15/// operations that read the uninitialized values, or form references to them, either directly or
16/// indirectly through any of the numerical library routines, unless it is explicitly permitted
17pub struct RowRef<'a, T, Cols = usize, CStride = isize> {
18	pub(crate) trans: ColRef<'a, T, Cols, CStride>,
19}

58	/// creates a `RowRef` from pointers to the column vector data, number of rows, and row stride
59	///
60	/// # safety
61	/// this function has the same safety requirements as
62	/// [`MatRef::from_raw_parts(ptr, 1, ncols, 0, col_stride)`]
63	#[inline(always)]
64	#[track_caller]
65	pub unsafe fn from_raw_parts(ptr: *const T, ncols: Cols, col_stride: CStride) -> Self {
66		Self {
67			trans: ColRef::from_raw_parts(ptr, ncols, col_stride),
68		}
69	}

107	/// returns a raw pointer to the element at the given index, assuming the provided index
108	/// is within the row bounds
109	///
110	/// # safety
111	/// the behavior is undefined if any of the following conditions are violated:
112	/// * `col < self.ncols()`
113	#[inline(always)]
114	#[track_caller]
115	pub unsafe fn ptr_inbounds_at(&self, col: Idx<Cols>) -> *const T {
116		debug_assert!(all(col < self.ncols()));
117		self.trans.ptr_inbounds_at(col)
118	}
119
120	/// splits the row vertically at the given column into two parts and returns an array of
121	/// each subrow, in the following order:
122	/// * left
123	/// * right
124	///
125	/// # panics
126	/// the function panics if the following condition is violated:
127	/// * `col <= self.ncols()`
128	#[inline]
129	#[track_caller]
130	pub fn split_at_col(self, col: IdxInc<Cols>) -> (RowRef<'a, T, usize, CStride>, RowRef<'a, T, usize, CStride>) {
131		assert!(all(col <= self.ncols()));
132		let rs = self.col_stride();
133
134		let top = self.as_ptr();
135		let bot = self.ptr_at(col);
136		unsafe {
137			(
138				RowRef::from_raw_parts(top, col.unbound(), rs),
139				RowRef::from_raw_parts(bot, self.ncols().unbound() - col.unbound(), rs),
140			)
141		}
142	}

232	/// returns a view over the subrow starting at column `col_start`, and with number of
233	/// columns `ncols`
234	///
235	/// # panics
236	/// the function panics if any of the following conditions are violated:
237	/// * `col_start <= self.ncols()`
238	/// * `ncols <= self.ncols() - col_start`
239	#[inline]
240	pub fn subcols<V: Shape>(self, col_start: IdxInc<Cols>, ncols: V) -> RowRef<'a, T, V, CStride> {
241		assert!(all(col_start <= self.ncols()));
242		{
243			let ncols = ncols.unbound();
244			let full_ncols = self.ncols().unbound();
245			let col_start = col_start.unbound();
246			assert!(all(ncols <= full_ncols - col_start));
247		}
248		let cs = self.col_stride();
249		unsafe { RowRef::from_raw_parts(self.ptr_at(col_start), ncols, cs) }
250	}

430impl<'COLS, 'a, T, CStride: Stride> RowRef<'a, T, Dim<'COLS>, CStride> {
431	#[doc(hidden)]
432	#[inline]
433	pub fn split_cols_with<'LEFT, 'RIGHT>(
434		self,
435		col: Partition<'LEFT, 'RIGHT, 'COLS>,
436	) -> (RowRef<'a, T, Dim<'LEFT>, CStride>, RowRef<'a, T, Dim<'RIGHT>, CStride>) {
437		let (a, b) = self.split_at_col(col.midpoint());
438		(a.as_col_shape(col.head), b.as_col_shape(col.tail))
439	}
440}

442impl<T: core::fmt::Debug, Cols: Shape, CStride: Stride> core::fmt::Debug for RowRef<'_, T, Cols, CStride> {
443	fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
444		fn imp<T: core::fmt::Debug>(f: &mut core::fmt::Formatter<'_>, this: RowRef<'_, T, Dim<'_>>) -> core::fmt::Result {
445			f.debug_list()
446				.entries(this.ncols().indices().map(|j| crate::hacks::hijack_debug(this.at(j))))
447				.finish()
448		}
449
450		with_dim!(N, self.ncols().unbound());
451		imp(f, self.as_col_shape(N).as_dyn_stride())
452	}
