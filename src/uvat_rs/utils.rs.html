30/// Apply constraints on matrix `l` and right hand side column vector `rhs`.
31/// Constraints are given by a list of indices `c_indices` and the values `c_values` :
32/// in the result, the values at index `c_index[i]` will equal `c_value[i]`.
33pub fn apply_constraints<I: Index>(
34    l: &mut SparseColMat<I, f64>,
35    mut rhs: ColMut<f64>,
36    c_indices: &[usize],
37    c_values: &[f64],
38) {
39    {
40        let (sym, v) = l.parts_mut();
41        let indices = sym.row_idx();
42        indices
43            .par_iter()
44            .zip(v.par_iter_mut())
45            .for_each(|(i, value)| {
46                if c_indices.contains(&i.zx()) {
47                    *value = 0.;
48                }
49            });
50    }
51    for (index, &c_value) in c_indices.iter().zip(c_values) {
52        let (sym, v) = l.parts_mut();
53        let indices = sym.row_idx_of_col_raw(*index);
54        let range = sym.col_range(*index);
55        for (value, i) in v[range].iter_mut().zip(indices) {
56            if i.zx() != *index {
57                rhs[i.zx()] -= *value * c_value;
58                *value = 0.;
59            } else {
60                rhs[i.zx()] = c_value;
61                *value = 1.;
62            }
63        }
64    }
65}

67/// Same as [`apply_constraints`] but with a matrix for rhs and rows for constraints values.
68pub fn apply_constraints_mat<I: Index>(
69    l: &mut SparseColMat<I, f64>,
70    mut rhs: MatMut<f64>,
71    c_indices: &[I],
72    c_values: &[Row<f64>],
73) {
74    let (sym, v) = l.parts_mut();
75    let indices = sym.row_idx();
76    indices
77        .par_iter()
78        .zip(v.par_iter_mut())
79        .for_each(|(i, value)| {
80            if c_indices.contains(&i) {
81                *value = 0.;
82            }
83        });
84    for (index, c_value) in c_indices.iter().zip(c_values) {
85        let (sym, v) = l.parts_mut();
86        let indices = sym.row_idx_of_col_raw(index.zx());
87        let range = sym.col_range(index.zx());
88        for (value, i) in v[range].iter_mut().zip(indices) {
89            if *i != *index {
90                rhs.rb_mut()
91                    .row_mut(i.zx())
92                    .iter_mut()
93                    .zip(c_value.iter())
94                    .for_each(|(v, c)| *v -= *value * c);
95                *value = 0.;
96            } else {
97                rhs.rb_mut()
98                    .row_mut(i.zx())
99                    .iter_mut()
100                    .zip(c_value.iter())
101                    .for_each(|(v, c)| *v = *c);
102                *value = 1.;
103            }
104        }
105    }
106}

108/// Move the vertices from `src` by at most `delta_dst` while avoiding triangle flips and also minimizing the energy evaluted by `e`
109pub fn flip_avoiding_line_search<I: Index, F: Fn(&[[f64; 2]]) -> f64>(
110    src: &mut [[f64; 2]],
111    delta_dst: &[[f64; 2]],
112    f: &[[I; 3]],
113    e: F,
114) -> f64 {
115    let mut t = 1.;
116    let start_e = e(src);
117    //TODO parallelize
118    for face in f {
119        let v1 = src[face[0].zx()];
120        let v1p = delta_dst[face[0].zx()];
121        let v2 = src[face[1].zx()];
122        let v2p = delta_dst[face[1].zx()];
123        let v3 = src[face[2].zx()];
124        let v3p = delta_dst[face[2].zx()];
125        let p1 = [v2[0] - v1[0], v2[1] - v1[1]];
126        let p1p = [v2p[0] - v1p[0], v2p[1] - v1p[1]];
127        let p2 = [v3[0] - v1[0], v3[1] - v1[1]];
128        let p2p = [v3p[0] - v1p[0], v3p[1] - v1p[1]];
129        let x1 = p1[0];
130        let y1 = p1[1];
131        let x1p = p1p[0];
132        let y1p = p1p[1];
133        let x2 = p2[0];
134        let y2 = p2[1];
135        let x2p = p2p[0];
136        let y2p = p2p[1];
137        let c = x1 * y2 - y1 * x2;
138        let b = x1p * y2 + x1 * y2p - y1p * x2 - y1 * x2p;
139        let a = x1p * y2p - y1p * x2p;
140        let delta = b.powi(2) - 4. * a * c;
141        if delta < 0. {
142            continue;
143        }
144        let t1 = (-b - delta.sqrt()) / (2. * a);
145        let t2 = (-b + delta.sqrt()) / (2. * a);
146        if t1 > 0. && t1 < t {
147            t = t1 * 0.9;
148        }
149        if t2 > 0. && t2 < t {
150            t = t2 * 0.9;
151        }
152    }
153    for _ in 0..10 {
154        let mut temp_res = src.to_owned();
155        for (src_row, t_row) in temp_res.iter_mut().zip(delta_dst) {
156            src_row[0] += t * t_row[0];
157            src_row[1] += t * t_row[1];
158        }
159        let cur_e = e(&temp_res);
160        if cur_e < start_e {
161            src.copy_from_slice(&temp_res);
162            return cur_e;
163        }
164        t *= 0.5;
165    }
166    start_e
167}

169/// Build an edge map from faces `f` and total the number of vertices `nv`.
170///
171/// The result is given as a list of elements corresponding to each face, where each element has 3 indices values corresponding
172/// to the adjacent face for the edge, and an number indicating the index in which the edge is also found in the opposite face
173/// (negative values are used to indicate border edges).
174///
175/// The first value corresponds to the edge given by vertices `0` and `1`, the second by vertices `1` and `2` et the third `2` and `0`.
176// Relies on atomics, however only load and stores are used, no memory lock should happen.
177pub fn build_edge_map<
178    I: Index
179        + std::ops::Shl<usize, Output = I>
180        + std::ops::Shr<usize, Output = I>
181        + std::ops::BitAnd<Output = I>,
182>(
183    f: &[[I; 3]],
184    nv: usize,
185) -> Vec<([I; 3], [i8; 3])> {
186    let arch = Arch::new();
187    arch.dispatch(|| {
188        let mut faces_deg = vec![0_u8; nv + 1];
189        let mut faces_offsets = vec![0_u8; 3 * f.len()];
190        for (row, off) in f.into_iter().zip(faces_offsets.chunks_exact_mut(3)) {
191            if row[0] < row[1] {
192                off[0] = faces_deg[row[0].zx()] as u8;
193                faces_deg[row[0].zx()] += 1;
194            } else {
195                off[0] = faces_deg[row[1].zx()] as u8;
196                faces_deg[row[1].zx()] += 1;
197            }
198            if row[1] < row[2] {
199                off[1] = faces_deg[row[1].zx()] as u8;
200                faces_deg[row[1].zx()] += 1;
201            } else {
202                off[1] = faces_deg[row[2].zx()] as u8;
203                faces_deg[row[2].zx()] += 1;
204            }
205            if row[0] < row[2] {
206                off[2] = faces_deg[row[0].zx()] as u8;
207                faces_deg[row[0].zx()] += 1;
208            } else {
209                off[2] = faces_deg[row[2].zx()] as u8;
210                faces_deg[row[2].zx()] += 1;
211            }
212        }
213
214        let mut offset = I::truncate(0);
215        let faces_deg: Vec<_> = faces_deg
216            .into_iter()
217            .map(|v| {
218                let value = I::truncate(v as usize);
219                offset += value;
220                offset - value
221            })
222            .collect();
223
224        let edges_to_faces: Vec<_> =
225            std::iter::repeat_with(|| (AtomicU32::new(0), AtomicU32::new(0)))
226                .take(offset.zx())
227                .collect();
228        let cur_offset = faces_deg;
229
230        f.into_par_iter()
231            .enumerate()
232            .zip(faces_offsets.par_chunks_exact(3))
233            .for_each(|((i, row), f_off)| {
234                if row[0] < row[1] {
235                    edges_to_faces[cur_offset[row[0].zx()].zx() + f_off[0] as usize]
236                        .0
237                        .store(
238                            (row[1].zx() << 1) as u32,
239                            std::sync::atomic::Ordering::Relaxed,
240                        );
241                    edges_to_faces[cur_offset[row[0].zx()].zx() + f_off[0] as usize]
242                        .1
243                        .store((i << 1) as u32, std::sync::atomic::Ordering::Relaxed);
244                } else {
245                    edges_to_faces[cur_offset[row[1].zx()].zx() + f_off[0] as usize]
246                        .0
247                        .store(
248                            (row[0].zx() << 1) as u32,
249                            std::sync::atomic::Ordering::Relaxed,
250                        );
251                    edges_to_faces[cur_offset[row[1].zx()].zx() + f_off[0] as usize]
252                        .1
253                        .store((i << 1) as u32, std::sync::atomic::Ordering::Relaxed);
254                }
255                if row[1] < row[2] {
256                    edges_to_faces[cur_offset[row[1].zx()].zx() + f_off[1] as usize]
257                        .0
258                        .store(
259                            (row[2].zx() << 1) as u32,
260                            std::sync::atomic::Ordering::Relaxed,
261                        );
262                    edges_to_faces[cur_offset[row[1].zx()].zx() + f_off[1] as usize]
263                        .1
264                        .store(((i << 1) + 1) as u32, std::sync::atomic::Ordering::Relaxed);
265                } else {
266                    edges_to_faces[cur_offset[row[2].zx()].zx() + f_off[1] as usize]
267                        .0
268                        .store(
269                            (row[1].zx() << 1) as u32,
270                            std::sync::atomic::Ordering::Relaxed,
271                        );
272                    edges_to_faces[cur_offset[row[2].zx()].zx() + f_off[1] as usize]
273                        .1
274                        .store(((i << 1) + 1) as u32, std::sync::atomic::Ordering::Relaxed);
275                }
276                if row[0] < row[2] {
277                    edges_to_faces[cur_offset[row[0].zx()].zx() + f_off[2] as usize]
278                        .0
279                        .store(
280                            ((row[2].zx() << 1) + 1) as u32,
281                            std::sync::atomic::Ordering::Relaxed,
282                        );
283                    edges_to_faces[cur_offset[row[0].zx()].zx() + f_off[2] as usize]
284                        .1
285                        .store((i << 1) as u32, std::sync::atomic::Ordering::Relaxed);
286                } else {
287                    edges_to_faces[cur_offset[row[2].zx()].zx() + f_off[2] as usize]
288                        .0
289                        .store(
290                            ((row[0].zx() << 1) + 1) as u32,
291                            std::sync::atomic::Ordering::Relaxed,
292                        );
293                    edges_to_faces[cur_offset[row[2].zx()].zx() + f_off[2] as usize]
294                        .1
295                        .store((i << 1) as u32, std::sync::atomic::Ordering::Relaxed);
296                }
297            });
298        let edges: Vec<_> = vec![([0; 3], [-1; 3]); f.len()]
299            .into_iter()
300            .map(|(v1, v2)| {
301                (
302                    [
303                        AtomicU32::from(v1[0]),
304                        AtomicU32::from(v1[1]),
305                        AtomicU32::from(v1[2]),
306                    ],
307                    [
308                        AtomicI8::from(v2[0]),
309                        AtomicI8::from(v2[1]),
310                        AtomicI8::from(v2[2]),
311                    ],
312                )
313            })
314            .collect();
315
316        let mut edges_to_faces: Vec<_> = edges_to_faces
317            .into_iter()
318            .map(|(e1, e2)| (e1.into_inner(), e2.into_inner()))
319            .collect();
320
321        rayon::iter::split(
322            SubSlices {
323                idx: &cur_offset,
324                data: &mut edges_to_faces,
325            },
326            SubSlices::splitter,
327        )
328        .for_each(|s| {
329            let r = s.idx[0];
330            for offs in s.idx.windows(2) {
331                let slice = &mut s.data[(offs[0] - r).zx()..(offs[1] - r).zx()];
332                slice.sort_unstable_by_key(|item| item.0);
333                let mut i = 0;
334                while slice.len() > 0 && i < (slice.len() - 1) {
335                    let item1 = slice[i];
336                    let item2 = slice[i + 1];
337                    if item1.0 >> 1 == item2.0 >> 1 {
338                        let f1 = item1.1 >> 1;
339                        let f2 = item2.1 >> 1;
340                        let e_i_1 = 2 * (item1.0 & 1) + (item1.1 & 1);
341                        let e_i_2 = 2 * (item2.0 & 1) + (item2.1 & 1);
342                        unsafe {
343                            edges.get_unchecked(f1.zx()).0[e_i_1.zx() as usize]
344                                .store(f2.zx() as u32, std::sync::atomic::Ordering::Relaxed);
345                            edges.get_unchecked(f1.zx()).1[e_i_1.zx() as usize]
346                                .store(e_i_2.zx() as i8, std::sync::atomic::Ordering::Relaxed);
347                            edges.get_unchecked(f2.zx()).0[e_i_2.zx() as usize]
348                                .store(f1.zx() as u32, std::sync::atomic::Ordering::Relaxed);
349                            edges.get_unchecked(f2.zx()).1[e_i_2.zx() as usize]
350                                .store(e_i_1.zx() as i8, std::sync::atomic::Ordering::Relaxed);
351                        }
352                        i += 2;
353                    } else {
354                        i += 1;
355                    }
356                }
357            }
358        });
359        edges
360            .into_iter()
361            .map(|(e, e_i)| {
362                let [e0, e1, e2] = e;
363                let [e_i0, e_i1, e_i2] = e_i;
364                (
365                    [
366                        I::truncate(e0.into_inner() as usize),
367                        I::truncate(e1.into_inner() as usize),
368                        I::truncate(e2.into_inner() as usize),
369                    ],
370                    [e_i0.into_inner(), e_i1.into_inner(), e_i2.into_inner()],
371                )
372            })
373            .collect()
374    })
375}

377struct SubSlices<'a, 'b, T, I: Index> {
378    idx: &'a [I],
379    data: &'b mut [T],
380}

382impl<'a, 'b, T, I: Index> SubSlices<'a, 'b, T, I> {
383    fn splitter(self) -> (Self, Option<Self>) {
384        if self.idx.len() <= 2 {
385            return (self, None);
386        }
387        let mid = self.idx.len() / 2;
388        let (idx_r, idx_l) = (&self.idx[0..mid + 1], &self.idx[mid..]);
389        let (data_r, data_l) = self.data.split_at_mut(idx_l[0].zx() - idx_r[0].zx());
390        (
391            Self {
392                idx: idx_r,
393                data: data_r,
394            },
395            Some(Self {
396                idx: idx_l,
397                data: data_l,
398            }),
399        )
400    }
401}

403fn get_next_boundary_vertex<I: Index>(
404    index: &mut I,
405    vertex_index: &mut u8,
406    f: &[[I; 3]],
407    e: &[([I; 3], [i8; 3])],
408    marked: &mut [bool],
409) -> I {
410    if e[index.zx()].1[*vertex_index as usize] < 0 {
411        *vertex_index = (*vertex_index + 1) % 3;
412        f[index.zx()][*vertex_index as usize]
413    } else {
414        loop {
415            let tmp_vert = (e[index.zx()].1[*vertex_index as usize] + 1) % 3;
416            let tmp_f = e[index.zx()].0[*vertex_index as usize];
417            *vertex_index = tmp_vert as u8;
418            *index = tmp_f;
419            marked[index.zx()] = true;
420            if e[index.zx()].1[*vertex_index as usize] < 0 {
421                *vertex_index = (*vertex_index + 1) % 3;
422                return f[index.zx()][*vertex_index as usize];
423            }
424        }
425    }
426}

428/// Find all ordered boundary loops from face indices `f` and edge map `e`.
429pub fn get_boundary_loop<I: Index>(f: &[[I; 3]], e: &[([I; 3], [i8; 3])]) -> Vec<Vec<I>> {
430    let mut res = Vec::new();
431    let mut marked = vec![false; e.len()];
432    for (i, (face, (_, adj))) in f.into_iter().zip(e.iter()).enumerate() {
433        if !marked[i] && (adj[0] < 0 || adj[1] < 0 || adj[2] < 0) {
434            marked[i] = true;
435            let (start_vertex, mut cur_index) = if adj[0] < 0 {
436                (face[0], 1)
437            } else if adj[1] < 0 {
438                (face[1], 2)
439            } else {
440                (face[2], 0)
441            };
442            let mut cur_face = I::truncate(i);
443            let mut temp_res = vec![start_vertex, face[cur_index as usize]];
444            loop {
445                let new_vertex: I =
446                    get_next_boundary_vertex(&mut cur_face, &mut cur_index, &f, &e, &mut marked);
447                if new_vertex == start_vertex {
448                    break;
449                }
450                temp_res.push(new_vertex);
451            }
452            res.push(temp_res);
453        } else {
454            marked[i] = true;
455        }
456    }
457    res
458}

470/// Add length to all edge lengths if the triangle inequality isn't satisfied with enough margin
471/// for stability. From <https://www.cs.cmu.edu/~kmcrane/Projects/NonmanifoldLaplace/NonmanifoldLaplace.pdf>
472pub fn mollify(l: &mut [[f64; 3]], delta: f64) {
473    let arch = Arch::new();
474    arch.dispatch(|| {
475        let max: f64 = l
476            .par_iter()
477            .map(|row| {
478                let mut max: f64 = 0.;
479                for i in 0..3 {
480                    let l1 = row[i];
481                    let l2 = row[(i + 1) % 3];
482                    let l3 = row[(i + 2) % 3];
483                    max = max.max(delta - l1 - l2 + l3);
484                }
485                max
486            })
487            .reduce_with(|a, b| a.max(b))
488            .unwrap();
489        let epsilon = max.max(0.);
490
491        l.par_iter_mut().for_each(|l| {
492            l[0] += epsilon;
493            l[1] += epsilon;
494            l[2] += epsilon;
495        });
496    })
497}

499/// Returns wether the triangle pair with shaired edge `l_s`, shaire by triangle `1` ( `l_s`,
500/// `l11`, `l12`) and triangle `2` (`l_s`, `l21`, `l22`) is Delaunay.
501pub fn is_delaunay(l_s: f64, l11: f64, l12: f64, l21: f64, l22: f64) -> bool {
502    let tan_1 = f64::sqrt(
503        ((l_s - l11 + l12) * (l_s - l12 + l11)) / ((l_s + l11 + l12) * (-l_s + l11 + l12)),
504    );
505    let tan_2 = f64::sqrt(
506        ((l_s - l21 + l22) * (l_s - l22 + l21)) / ((l_s + l21 + l22) * (-l_s + l21 + l22)),
507    );
508    let cot_1 = (1. - tan_1 * tan_1) / (2. * tan_1);
509    let cot_2 = (1. - tan_2 * tan_2) / (2. * tan_2);
510    cot_1 + cot_2 >= 0.
511}

513/// Compute lengths, faces and edge map from the intrinsic Delaunay triangulations of the surface.
514///
515/// Arguments :
516/// * `v` : vertices of the surface
517/// * `f` : faces indices of the surface
518/// * `e` : edge map of the surface, as computed by [`build_edge_map`]
519pub fn build_intrinsic_delaunay<
520    I: Index
521        + std::ops::Shl<usize, Output = I>
522        + std::ops::Shr<usize, Output = I>
523        + std::ops::BitAnd<Output = I>,
524>(
525    v: &[[f64; 3]],
526    mut f: Vec<[I; 3]>,
527    mut e: Vec<([I; 3], [i8; 3])>,
528) -> (Vec<[f64; 3]>, Vec<[I; 3]>, Vec<([I; 3], [i8; 3])>) {
529    struct Face<I> {
530        v: [I; 3],
531        adj_f: [I; 3],
532        adj_f_i: [i8; 3],
533        l: [f64; 3],
534    }
535
536    let arch = Arch::new();
537    let mut lengths = arch.dispatch(|| {
538        let mut edges_t = Vec::new();
539        f.par_iter()
540            .map(|row| {
541                let v1 = &v[row[0].zx()];
542                let v2 = &v[row[1].zx()];
543                let v3 = &v[row[2].zx()];
544                [
545                    v1[0] - v2[0],
546                    v1[1] - v2[1],
547                    v1[2] - v2[2],
548                    v3[0] - v2[0],
549                    v3[1] - v2[1],
550                    v3[2] - v2[2],
551                    v1[0] - v3[0],
552                    v1[1] - v3[1],
553                    v1[2] - v3[2],
554                ]
555            })
556            .collect_into_vec(&mut edges_t);
557        let edges_t = edges_t.into_flattened();
558        pulp::as_arrays::<9, f64>(&edges_t)
559            .0
560            .into_iter()
561            .map(|row| {
562                [
563                    f64::sqrt(row[0] * row[0] + row[1] * row[1] + row[2] * row[2]),
564                    f64::sqrt(row[3] * row[3] + row[4] * row[4] + row[5] * row[5]),
565                    f64::sqrt(row[6] * row[6] + row[7] * row[7] + row[8] * row[8]),
566                ]
567            })
568            .collect::<Vec<_>>()
569    });
570
571    mollify(&mut lengths, 10e-8);
572
573    let mut faces = Vec::new();
574    f.par_iter()
575        .zip(lengths.par_iter())
576        .zip(e.par_iter())
577        .map(|((f, l), e)| Face {
578            v: [f[0], f[1], f[2]],
579            adj_f: e.0,
580            adj_f_i: e.1,
581            l: *l,
582        })
583        .collect_into_vec(&mut faces);
584
585    let mut edges_queue = VecDeque::with_capacity(faces.len());
586    let mut edges_marked = vec![false; 3 * faces.len()];
587
588    for (face_index, face) in faces.iter().enumerate() {
589        for i in 0..3 {
590            if face.adj_f_i[i] >= 0 && face_index < face.adj_f[i].zx() {
591                edges_queue.push_back((I::truncate(face_index), i as u8));
592                edges_marked[3 * face_index + i] = true;
593            }
594        }
595    }
596
597    while let Some((face_index, edge_index)) = edges_queue.pop_back() {
598        edges_marked[3 * face_index.zx() + edge_index as usize] = false;
599        let f1 = &faces[face_index.zx()];
600        let opp_f = f1.adj_f[edge_index as usize];
601        let opp_f_i = f1.adj_f_i[edge_index as usize];
602        if opp_f_i >= 0 && face_index != opp_f {
603            let e_i_1 = edge_index;
604            let e_i_2 = if edge_index + 1 < 3 {
605                edge_index + 1
606            } else {
607                edge_index - 2
608            };
609            let e_i_3 = if edge_index + 2 < 3 {
610                edge_index + 2
611            } else {
612                edge_index - 1
613            };
614            let l_s = f1.l[e_i_1 as usize];
615            let l11 = f1.l[e_i_2 as usize];
616            let l12 = f1.l[e_i_3 as usize];
617            let f2 = &faces[opp_f.zx()];
618            let oe_i_1 = opp_f_i as usize;
619            let oe_i_2 = if opp_f_i + 1 < 3 {
620                oe_i_1 + 1
621            } else {
622                oe_i_1 - 2
623            };
624            let oe_i_3 = if opp_f_i + 2 < 3 {
625                oe_i_1 + 2
626            } else {
627                oe_i_1 - 1
628            };
629            let l21 = f2.l[oe_i_2];
630            let l22 = f2.l[oe_i_3];
631            if !is_delaunay(l_s, l11, l12, l21, l22) {
632                let f11 = f1.adj_f[e_i_2 as usize];
633                let f12 = f1.adj_f[e_i_3 as usize];
634                let f21 = f2.adj_f[oe_i_2 as usize];
635                let f22 = f2.adj_f[oe_i_3 as usize];
636                let i11 = f1.adj_f_i[e_i_2 as usize];
637                let i12 = f1.adj_f_i[e_i_3 as usize];
638                let i21 = f2.adj_f_i[oe_i_2 as usize];
639                let i22 = f2.adj_f_i[oe_i_3 as usize];
640                let v11 = f1.v[e_i_2 as usize];
641                let v12 = f1.v[e_i_3 as usize];
642                let v21 = f2.v[oe_i_2 as usize];
643                let v22 = f2.v[oe_i_3 as usize];
644                let tan_a_2 = f64::sqrt(
645                    ((l12 - l11 + l_s) * (l12 - l_s + l11))
646                        / ((l12 + l11 + l_s) * (l_s + l11 - l12)),
647                );
648                let tan_d_2 = f64::sqrt(
649                    ((l21 - l22 + l_s) * (l21 - l_s + l22))
650                        / ((l21 + l22 + l_s) * (l_s + l22 - l21)),
651                );
652                let tan_s = (tan_a_2 + tan_d_2) / (1. - tan_a_2 * tan_d_2);
653                let cos = (1. - tan_s * tan_s) / (1. + tan_s * tan_s);
654                let new_l = f64::sqrt(l11 * l11 + l22 * l22 - 2. * l11 * l22 * cos);
655                let mut new_f_1 = match e_i_1 {
656                    2 => Face {
657                        v: [v11, v12, v22],
658                        l: [l11, new_l, l22],
659                        adj_f: [f11, opp_f, f22],
660                        adj_f_i: [i11, oe_i_3 as i8, i22],
661                    },
662                    1 => Face {
663                        v: [v12, v22, v11],
664                        l: [new_l, l22, l11],
665                        adj_f: [opp_f, f22, f11],
666                        adj_f_i: [oe_i_3 as i8, i22, i11],
667                    },
668                    _ => Face {
669                        v: [v22, v11, v12],
670                        l: [l22, l11, new_l],
671                        adj_f: [f22, f11, opp_f],
672                        adj_f_i: [i22, i11, oe_i_3 as i8],
673                    },
674                };
675                let mut new_f_2 = match oe_i_1 {
676                    2 => Face {
677                        v: [v21, v22, v12],
678                        l: [l21, new_l, l12],
679                        adj_f: [f21, face_index, f12],
680                        adj_f_i: [i21, e_i_3 as i8, i12],
681                    },
682                    1 => Face {
683                        v: [v22, v12, v21],
684                        l: [new_l, l12, l21],
685                        adj_f: [face_index, f12, f21],
686                        adj_f_i: [e_i_3 as i8, i12, i21],
687                    },
688                    _ => Face {
689                        v: [v12, v21, v22],
690                        l: [l12, l21, new_l],
691                        adj_f: [f12, f21, face_index],
692                        adj_f_i: [i12, i21, e_i_3 as i8],
693                    },
694                };
695                if f22 == face_index {
696                    new_f_1.adj_f_i[e_i_1 as usize] = e_i_1 as i8;
697                } else if f22 == opp_f {
698                    new_f_1.adj_f_i[e_i_1 as usize] = e_i_1 as i8;
699                    new_f_1.adj_f[e_i_1 as usize] = face_index;
700                }
701                if f11 == face_index {
702                    new_f_1.adj_f_i[e_i_2 as usize] = e_i_2 as i8;
703                } else if f11 == opp_f {
704                    new_f_1.adj_f_i[e_i_2 as usize] = e_i_2 as i8;
705                    new_f_1.adj_f[e_i_2 as usize] = face_index;
706                }
707                if f12 == opp_f {
708                    new_f_2.adj_f_i[oe_i_1 as usize] = oe_i_1 as i8;
709                } else if f12 == face_index {
710                    new_f_2.adj_f_i[oe_i_1 as usize] = oe_i_1 as i8;
711                    new_f_2.adj_f[oe_i_1 as usize] = opp_f;
712                }
713                if f21 == opp_f {
714                    new_f_2.adj_f_i[oe_i_2 as usize] = oe_i_2 as i8;
715                } else if f21 == face_index {
716                    new_f_2.adj_f_i[oe_i_2 as usize] = oe_i_2 as i8;
717                    new_f_2.adj_f[oe_i_2] = opp_f;
718                }
719
720                faces[face_index.zx()] = new_f_1;
721                faces[opp_f.zx()] = new_f_2;
722
723                if i11 >= 0 && f11 != face_index && f11 != opp_f {
724                    let edge = if f11 < face_index {
725                        (f11, i11 as u8)
726                    } else {
727                        (face_index, e_i_2)
728                    };
729                    if !edges_marked[3 * edge.0.zx() + edge.1 as usize] {
730                        edges_marked[3 * edge.0.zx() + edge.1 as usize] = true;
731                        edges_queue.push_back(edge);
732                    }
733                }
734                if i22 >= 0 && f22 != face_index && f22 != opp_f {
735                    faces[f22.zx()].adj_f[i22 as usize] = face_index;
736                    faces[f22.zx()].adj_f_i[i22 as usize] = e_i_1 as i8;
737                    let edge = if f22 < face_index {
738                        (f22, i22 as u8)
739                    } else {
740                        (face_index, e_i_1)
741                    };
742                    if !edges_marked[3 * edge.0.zx() + edge.1 as usize] {
743                        edges_marked[3 * edge.0.zx() + edge.1 as usize] = true;
744                        edges_queue.push_back(edge);
745                    }
746                }
747                if i12 >= 0 && f12 != face_index && f12 != opp_f {
748                    faces[f12.zx()].adj_f[i12 as usize] = opp_f;
749                    faces[f12.zx()].adj_f_i[i12 as usize] = oe_i_1 as i8;
750                    let edge = if f12 < opp_f {
751                        (f12, i12 as u8)
752                    } else {
753                        (opp_f, oe_i_1 as u8)
754                    };
755                    if !edges_marked[3 * edge.0.zx() + edge.1 as usize] {
756                        edges_marked[3 * edge.0.zx() + edge.1 as usize] = true;
757                        edges_queue.push_back(edge);
758                    }
759                }
760                if i21 >= 0 && f21 != face_index && f21 != opp_f {
761                    let edge = if f21 < opp_f {
762                        (f21, i21 as u8)
763                    } else {
764                        (opp_f, oe_i_2 as u8)
765                    };
766                    if !edges_marked[3 * edge.0.zx() + edge.1 as usize] {
767                        edges_marked[3 * edge.0.zx() + edge.1 as usize] = true;
768                        edges_queue.push_back(edge);
769                    }
770                }
771            }
772        }
773    }
774
775    faces
776        .into_par_iter()
777        .zip(lengths.par_iter_mut())
778        .zip(f.par_iter_mut())
779        .zip(e.par_iter_mut())
780        .for_each(|(((face, l), f), e)| {
781            *l = face.l;
782            f[0] = face.v[0];
783            f[1] = face.v[1];
784            f[2] = face.v[2];
785            e.0 = face.adj_f;
786            e.1 = face.adj_f_i;
787        });
788    (lengths, f, e)
789}

791/// Build mass and stiffness matrices for the intrinsic Laplacian, from the intrinsic Delaunay triangulation
792/// (as precomputed by [`build_intrinsic_delaunay`]).
793///
794/// Slightly overkill for Tutte parameterization.
795pub fn build_mass_cotan_laplacian_intrinsic<I: Index>(
796    lengths: &[[f64; 3]],
797    f: &[[I; 3]],
798    e: &[([I; 3], [i8; 3])],
799    nv: usize,
800) -> (SparseColMat<I, f64>, SparseColMat<I, f64>) {
801    let mut cots = Vec::new();
802    lengths
803        .par_iter()
804        .map(|l| {
805            let l1 = l[1];
806            let l2 = l[2];
807            let l3 = l[0];
808            let l12 = l1 * l1;
809            let l22 = l2 * l2;
810            let l32 = l3 * l3;
811            let s = (l1 + l2 + l3) / 2.;
812            let a = f64::sqrt(s * (s - l1) * (s - l2) * (s - l3));
813            let cotan1 = (l22 + l32 - l12) / (8. * a);
814            let cotan2 = (l12 + l32 - l22) / (8. * a);
815            let cotan3 = (l22 + l12 - l32) / (8. * a);
816            [cotan1, cotan2, cotan3]
817        })
818        .collect_into_vec(&mut cots);
819
820    let mut sum_cots = Vec::new();
821    cots.par_iter()
822        .zip(e.into_par_iter())
823        .enumerate()
824        .map(|(i, (cot, e))| {
825            let mut cotan0 = cot[0];
826            let mut cotan1 = cot[1];
827            let mut cotan2 = cot[2];
828            if e.1[1] >= 0 && e.0[1].zx() > i {
829                cotan0 += cots[e.0[1].zx()][(e.1[1] as usize + 2) % 3];
830            }
831            if e.1[2] >= 0 && e.0[2].zx() > i {
832                cotan1 += cots[e.0[2].zx()][(e.1[2] as usize + 2) % 3];
833            }
834            if e.1[0] >= 0 && e.0[0].zx() > i {
835                cotan2 += cots[e.0[0].zx()][(e.1[0] as usize + 2) % 3];
836            }
837            [cotan0, cotan1, cotan2]
838        })
839        .collect_into_vec(&mut sum_cots);
840
841    let mut deg = vec![0_u8; nv + 1];
842    let mut faces_offsets = vec![0_u8; 6 * f.len()];
843    for (i, ((face, edge), off)) in f
844        .into_iter()
845        .zip(e)
846        .zip(faces_offsets.chunks_exact_mut(6))
847        .enumerate()
848    {
849        if edge.1[0] < 0 || edge.0[0].zx() > i {
850            deg[face[0].zx()] += 1;
851            deg[face[1].zx()] += 1;
852            off[0] = deg[face[0].zx()];
853            off[1] = deg[face[1].zx()];
854        }
855        if edge.1[1] < 0 || edge.0[1].zx() > i {
856            deg[face[1].zx()] += 1;
857            deg[face[2].zx()] += 1;
858            off[2] = deg[face[1].zx()];
859            off[3] = deg[face[2].zx()];
860        }
861        if edge.1[2] < 0 || edge.0[2].zx() > i {
862            deg[face[2].zx()] += 1;
863            deg[face[0].zx()] += 1;
864            off[4] = deg[face[2].zx()];
865            off[5] = deg[face[0].zx()];
866        }
867    }
868    let mut offset = 0_usize;
869    let offsets: Vec<I> = deg
870        .iter()
871        .map(|deg| {
872            offset += *deg as usize + 1;
873            I::truncate(offset - (*deg + 1) as usize)
874        })
875        .collect();
876
877    let indices: Vec<_> = vec![(0_u32, 0_u64); offset]
878        .into_iter()
879        .map(|(v1, v2)| (AtomicU32::from(v1 as u32), AtomicU64::from(v2)))
880        .collect();
881
882    f.into_par_iter()
883        .zip(sum_cots.into_par_iter())
884        .zip(faces_offsets.par_chunks_exact(6))
885        .for_each(|((face, cots), off)| {
886            let cotan0 = cots[0];
887            let cotan1 = cots[1];
888            let cotan2 = cots[2];
889            if off[0] > 0 {
890                indices[offsets[face[0].zx()].zx() + off[0] as usize]
891                    .0
892                    .store(face[1].zx() as u32, Ordering::Relaxed);
893                indices[offsets[face[0].zx()].zx() + off[0] as usize]
894                    .1
895                    .store((-cotan2).to_bits(), Ordering::Relaxed);
896
897                indices[offsets[face[1].zx()].zx() + off[1] as usize]
898                    .0
899                    .store(face[0].zx() as u32, Ordering::Relaxed);
900                indices[offsets[face[1].zx()].zx() + off[1] as usize]
901                    .1
902                    .store((-cotan2).to_bits(), Ordering::Relaxed);
903            }
904            if off[2] > 0 {
905                indices[offsets[face[1].zx()].zx() + off[2] as usize]
906                    .0
907                    .store(face[2].zx() as u32, Ordering::Relaxed);
908                indices[offsets[face[1].zx()].zx() + off[2] as usize]
909                    .1
910                    .store((-cotan0).to_bits(), Ordering::Relaxed);
911
912                indices[offsets[face[2].zx()].zx() + off[3] as usize]
913                    .0
914                    .store(face[1].zx() as u32, Ordering::Relaxed);
915                indices[offsets[face[2].zx()].zx() + off[3] as usize]
916                    .1
917                    .store((-cotan0).to_bits(), Ordering::Relaxed);
918            }
919            if off[4] > 0 {
920                indices[offsets[face[2].zx()].zx() + off[4] as usize]
921                    .0
922                    .store(face[0].zx() as u32, Ordering::Relaxed);
923                indices[offsets[face[2].zx()].zx() + off[4] as usize]
924                    .1
925                    .store((-cotan1).to_bits(), Ordering::Relaxed);
926
927                indices[offsets[face[0].zx()].zx() + off[5] as usize]
928                    .0
929                    .store(face[2].zx() as u32, Ordering::Relaxed);
930                indices[offsets[face[0].zx()].zx() + off[5] as usize]
931                    .1
932                    .store((-cotan1).to_bits(), Ordering::Relaxed);
933            }
934        });
935
936    offsets[..(offsets.len() - 1)]
937        .par_iter()
938        .enumerate()
939        .for_each(|(i, off)| {
940            indices[off.zx()].0.store(i as u32, Ordering::Relaxed);
941        });
942    let mut indices: Vec<_> = indices
943        .into_iter()
944        .map(|(i, v)| {
945            (
946                I::truncate(i.into_inner() as usize),
947                f64::from_bits(v.into_inner()),
948            )
949        })
950        .collect();
951
952    rayon::iter::split(
953        SubSlices {
954            idx: &offsets,
955            data: &mut indices,
956        },
957        SubSlices::splitter,
958    )
959    .for_each(|s| {
960        let r = s.idx[0];
961        for offs in s.idx.windows(2) {
962            let slice = &mut s.data[(offs[0] - r).zx()..(offs[1] - r).zx()];
963            slice[0].1 = -slice[1..].iter().fold(0., |acc, x| acc + x.1);
964            slice.sort_unstable_by_key(|item| item.0);
965        }
966    });
967    let (indices, coeffs): (Vec<_>, Vec<_>) = indices.into_iter().unzip();
968
969    let mut coeffs_m = vec![0.; nv];
970    for ((row, l), cot) in f.into_iter().zip(lengths.iter()).zip(cots) {
971        let i1 = row[0];
972        let i2 = row[1];
973        let i3 = row[2];
974        let l1 = l[1];
975        let l2 = l[2];
976        let l3 = l[0];
977        let l12 = l1 * l1;
978        let l22 = l2 * l2;
979        let l32 = l3 * l3;
980        let cotan1 = cot[0];
981        let cotan2 = cot[1];
982        let cotan3 = cot[2];
983        coeffs_m[i1.zx()] += (l22 * cotan2 + l32 * cotan3) / 4.;
984        coeffs_m[i2.zx()] += (l12 * cotan1 + l32 * cotan3) / 4.;
985        coeffs_m[i3.zx()] += (l12 * cotan1 + l22 * cotan2) / 4.;
986    }
987    let triplets_m: Vec<_> = coeffs_m
988        .into_iter()
989        .enumerate()
990        .map(|(i, value)| Triplet::new(I::truncate(i), I::truncate(i), value))
991        .collect();
992
993    let symbolic = unsafe { SymbolicSparseColMat::new_unchecked(nv, nv, offsets, None, indices) };
994    (
995        SparseColMat::<I, f64>::try_new_from_triplets(nv, nv, &triplets_m).unwrap(),
996        SparseColMat::<I, f64>::new(symbolic, coeffs),
997    )
998}

1000/// Compute Tutte parameterization with border `b` mapped to a circle.
1001///
1002/// Uses intrinsic Laplacian, which is overkill
1003///
1004/// Arguments :
1005/// * `v` : surface vertices
1006/// * `f` : surface face indices
1007/// * `e` : surface edge map as obtained from [`build_edge_map`]
1008/// * `b` : choosen boundary to map on the circle (can be obtained from [`get_boundary_loop`])
1009pub fn compute_tutte_parameterization<I: MyIndex>(
1010    v: &[[f64; 3]],
1011    f: &[[I; 3]],
1012    e: Vec<([I; 3], [i8; 3])>,
1013    b: &[I],
1014) -> Vec<[f64; 2]> {
1015    let (l, f2, e) = build_intrinsic_delaunay(&v, f.to_owned(), e);
1016    let tot_area = l.iter().fold(0., |acc, row| {
1017        let l1 = row[0];
1018        let l2 = row[1];
1019        let l3 = row[2];
1020        let s = (l1 + l2 + l3) * 0.5;
1021        acc + (s * (s - l1) * (s - l2) * (s - l3)).sqrt()
1022    });
1023
1024    let (_mass, mut lap) = build_mass_cotan_laplacian_intrinsic(&l, &f2, &e, v.len());
1025    let mut rhs = Mat::<f64>::zeros(v.len(), 2);
1026
1027    let c = map_to_circle(b.len());
1028    let rows: Vec<_> = c
1029        .into_iter()
1030        .map(|row| Row::from_fn(2, |i| row[i] * tot_area.sqrt() / PI.sqrt()))
1031        .collect();
1032
1033    apply_constraints_mat(&mut lap, rhs.as_mut(), &b, &rows);
1034    let llt = lap.sp_cholesky(faer::Side::Lower).unwrap();
1035    llt.solve_in_place(rhs.as_mut());
1036
1037    rhs.row_iter()
1038        .map(|row| [row[0] as f64, row[1] as f64])
1039        .collect()
1040}
